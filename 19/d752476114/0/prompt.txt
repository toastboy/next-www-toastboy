     ```typescript
         const fromGameDay = await gameDayService.get(gameDayId);
         if (!fromGameDay) return [];
         const today = new Date();
         const gameDays = (await gameDayService.getAll({ fromDate: fromGameDay.date }))
             .filter((gameDay) => gameDay.date <= today);
         if (gameDays.length === 0) return [];

         const allTimeOutcomes = await outcomeService.getAllForYear(0);
         const allTimePlayerRecords = await loadLatestPlayerRecordsBefore(0, gameDayId);
         const years = [...new Set(gameDays.map(gd => gd.date.getFullYear()))];
         const playerRecords: PlayerRecordType[] = [];

         for (const year of years) {
             const yearPlayerRecords = await loadLatestPlayerRecordsBefore(year, gameDayId);
             const yearOutcomes = await outcomeService.getAllForYear(year);
             for (const gameDay of gameDays) {
                 if (gameDay.date.getFullYear() !== year || gameDay.date > today) continue;
                 const gameDayOutcomes = await outcomeService.getByGameDay(gameDay.id);
                 await calculateYearPlayerRecords(year, yearOutcomes, yearPlayerRecords, gameDay, gameDayOutcomes, playerRecords);
                 await calculateYearPlayerRecords(0, allTimeOutcomes, allTimePlayerRecords, gameDay, gameDayOutcomes, playerRecords);
             }
         }
         return playerRecords;
     }
     ```

     - New module-level helper added (placed near `calculateYearPlayerRecords`):
     ```typescript
     async function loadLatestPlayerRecordsBefore(
         year: number,
         beforeGameDayId: number,
     ): Promise<Record<number, Partial<PlayerRecordType>>> {
         const map: Record<number, Partial<PlayerRecordType>> = {};
         const groups = await prisma.playerRecord.groupBy({
             by: ['playerId'],
             where: { year, gameDayId: { lt: beforeGameDayId } },
             _max: { gameDayId: true },
         });
         const conditions = groups
             .filter(g => g._max.gameDayId !== null)
             .map(g => ({ playerId: g.playerId, gameDayId: g._max.gameDayId! }));
         if (conditions.length > 0) {
             const records = await prisma.playerRecord.findMany({
                 where: { year, OR: conditions },
             });
             for (const record of records) {
                 map[record.playerId] = PlayerRecordWriteInputSchema.parse(record);
             }
         }
         return map;
     }
     ```

5. Problem Solving:
   - **Root cause of wrong ranks**: `upsertFromGameDay` called `upsertForGameDay(gameDay.id)` per game day. Each call started with empty `yearPlayerRecords`, so rankings were computed only among players who played on that specific day. The bug produced two interleaved rank sequences: correct ones from a prior full recompute, and wrong ones (1, 2, 3... among a small subset) from the per-day calls.
   - **Fix approach**: Initialize `yearPlayerRecords` and `allTimePlayerRecords` from DB before processing, using a `groupBy` (to find each player's most recent gameDayId before the rebuild point) then `findMany` (to load those specific records). This correctly includes all historical players in rank computations.
   - **DB recovery**: A full recompute via `upsertForGameDay()` (no args) is needed once to restore the DB to a clean state.
   - **Pending warning**: The ESLint `Unsafe assignment of an 'any' value` at line 982 of the spec file has not been fixed yet.

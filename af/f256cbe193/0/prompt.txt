3. Files and Code Sections:
   - **`src/lib/useHydrated.ts`** (NEW FILE — the core fix):
     ```ts
     import { useSyncExternalStore } from 'react';

     // useSyncExternalStore requires a subscribe function; since the client snapshot
     // never changes (always true), no subscriptions are needed.
     // eslint-disable-next-line @typescript-eslint/no-empty-function
     const subscribe = () => () => {};

     /**
      * Returns true only on the client after React has hydrated.
      * Uses useSyncExternalStore so the server snapshot (false) differs from the
      * client snapshot (true) without triggering a setState-in-effect lint error.
      */
     export function useHydrated(): boolean {
         return useSyncExternalStore(subscribe, () => true, () => false);
     }
     ```

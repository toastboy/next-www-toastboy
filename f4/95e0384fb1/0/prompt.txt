     // useSyncExternalStore requires a subscribe function; since the client snapshot
     // never changes (always true), no subscriptions are needed.
     // eslint-disable-next-line @typescript-eslint/no-empty-function
     const subscribe = () => () => {};

      * Uses useSyncExternalStore so the server snapshot (false) differs from the
